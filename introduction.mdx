## Quick Start



To get started, you can use 1Router via API like this:

<CodeGroup>

```typescript typescript
fetch("https://1router.ai/api/v1/chat/completions", {
  method: "POST",
  headers: {
    "Authorization": `Bearer ${1ROUTER_API_KEY}`,
    "HTTP-Referer": `${YOUR_SITE_URL}`, // Optional, for including your app on 1router.ai rankings.
    "X-Title": `${YOUR_SITE_NAME}`, // Optional. Shows in rankings on 1router.ai.
    "Content-Type": "application/json"
  },
  body: JSON.stringify({
    "model": "meta-llama/llama-3.1-405b-instruct",
    "messages": [
      {
        "role": "user",
        "content": "What is the meaning of life?"
      }
    ]
  })
});

```

```python python
import requests
import json

response = requests.post(
    url="https://1router.ai/api/v1/chat/completions",
    headers={
        "Authorization": f"Bearer {1ROUTER_API_KEY}",
        "HTTP-Referer": f"{YOUR_SITE_URL}",  # Optional, for including your app on 1router.ai rankings.
        "X-Title": f"{YOUR_APP_NAME}",  # Optional. Shows in rankings on 1router.ai.
        "Content-Type": "application/json"
    },
    data=json.dumps({
        "model": "meta-llama/llama-3.1-405b-instruct",
        "messages": [
            {
                "role": "user",
                "content": "What is the meaning of life?"
            }
        ]
    })
)

```

```ruby ruby
require "open_router"

OpenRouter.configure do |config|
  config.access_token = ENV["ACCESS_TOKEN"]
  config.site_name = "YOUR_APP_NAME"
  config.site_url = "YOUR_SITE_URL"
end

OpenRouter::Client.new.complete(
  model: "meta-llama/llama-3.1-405b-instruct",
  messages: [
    {
      "role": "user",
      "content": "What is the meaning of life?"
    }
  ]
).then do |response|
  puts response.dig("choices", 0, "message", "content")
end


```

You can also use 1Router with OpenAI's client API:

<CodeGroup>

```typescript typescript
import OpenAI from "openai";

const openai = new OpenAI({
  baseURL: "https://1router.ai/api/v1",
  apiKey: $1ROUTER_API_KEY,
  defaultHeaders: {
    "HTTP-Referer": $YOUR_SITE_URL, // Optional, for including your app on 1router.ai rankings.
    "X-Title": $YOUR_SITE_NAME, // Optional. Shows in rankings on 1router.ai.
  }
});

async function main() {
  const completion = await openai.chat.completions.create({
    model: "meta-llama/llama-3.1-405b-instruct",
    messages: [
      {
        "role": "user",
        "content": "What is the meaning of life?"
      }
    ]
  });

  console.log(completion.choices[0].message);
}

main();


```

```python python
from openai import OpenAI
from os import getenv

# gets API Key from environment variable OPENROUTER_API_KEY
client = OpenAI(
    base_url="https://1router.ai/api/v1",
    api_key=getenv("1ROUTER_API_KEY"),
)

completion = client.chat.completions.create(
    extra_headers={
        "HTTP-Referer": YOUR_SITE_URL,  # Optional, for including your app on 1router.ai rankings.
        "X-Title": YOUR_APP_NAME,  # Optional. Shows in rankings on 1router.ai.
    },
    model="meta-llama/llama-3.1-405b-instruct",
    messages=[
        {
            "role": "user",
            "content": "What is the meaning of life?"
        }
    ]
)

print(completion.choices[0].message.content)

```
